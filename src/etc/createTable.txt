CREATE TABLE Agents (id serial primary key,
	stateDefinition varchar(80),
	actionSet varchar(80),
	rewardFunction varchar(80),
	acceptance real,
	targetDiameter real,
	maxStepsPerEpisode integer,
	successBounty real,
	failurePenalty real,
	device varchar(80),
	BATCH_SIZE integer,
	GAMMA real,
	TARGET_UPDATE integer,
	EPS_START real,
	EPS_END real,
	EPS_DECAY integer,
	MEMORY_SIZE integer,
	algorithm varchar(80),
	network varchar(255),
	optimizer varchar(80),
	stepSize real,
	trainEpisodes integer,
	trainBlob bytea
);
